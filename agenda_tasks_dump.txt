"use client";

import { useMemo, useState } from "react";
import type { Task } from "@/mock/tasks";
import { mockTasks } from "@/mock/tasks";
import { filterTasksByView, formatPeriodLabel } from "@/utils/tasks-time-filter";
import { AgendaViewSwitcher } from "@/components/agenda/AgendaViewSwitcher";
import { TaskCard } from "@/components/agenda/TaskCard";

export default function AgendaPage() {
  const [view, setView] = useState<"day" | "week" | "month">("day");
  const [referenceDate] = useState(new Date());
  const [tasks, setTasks] = useState<Task[]>(mockTasks);

  const periodLabel = useMemo(
    () => formatPeriodLabel(view, referenceDate),
    [view, referenceDate]
  );

  const visible = useMemo(
    () => filterTasksByView(tasks, view, referenceDate),
    [tasks, view, referenceDate]
  );

  return (
    <section className="space-y-4">
      <header className="flex items-start sm:items-center justify-between gap-3">
        <div>
          <h1 className="text-lg font-semibold text-zinc-900">
            Agenda
          </h1>
          <p className="text-xs text-zinc-500">
            {periodLabel}
          </p>
        </div>

        <AgendaViewSwitcher value={view} onChange={setView} />
      </header>

      {visible.length === 0 && (
        <p className="text-sm text-zinc-500">
          Nenhuma tarefa neste período.
        </p>
      )}

      <ul className="space-y-2">
        {visible.map((task) => (
          <TaskCard
            key={task.id}
            task={task}
            onChange={(next) => {
              setTasks((prev) =>
                prev.map((t) => (t.id === next.id ? next : t))
              );
            }}
          />
        ))}
      </ul>
    </section>
  );
}

import type { Task, TaskActionLog, TaskStatus } from "@/mock/tasks";

function nowIso() {
  return new Date().toISOString();
}

function appendAction(task: Task, action: TaskActionLog): Task {
  return {
    ...task,
    actions: [...(task.actions ?? []), action],
  };
}

export function markDone(task: Task, comment?: string | null): Task {
  if (task.status === "done") return task;

  const next: Task = {
    ...task,
    status: "done",
  };

  return appendAction(next, {
    type: "marked_done",
    at: nowIso(),
    comment: comment ?? null,
  });
}

export function reopen(task: Task, comment?: string | null): Task {
  if (task.status === "PLANNED") return task;

  const next: Task = {
    ...task,
    status: "PLANNED",
  };

  return appendAction(next, {
    type: "reopened",
    at: nowIso(),
    comment: comment ?? null,
  });
}

export function cancel(task: Task, comment?: string | null): Task {
  if (task.status === "canceled") return task;

  const next: Task = {
    ...task,
    status: "canceled",
  };

  return appendAction(next, {
    type: "canceled",
    at: nowIso(),
    comment: comment ?? null,
  });
}

export function statusLabel(status: TaskStatus): string {
  if (status === "PLANNED") return "Pendente";
  if (status === "done") return "Feita";
  return "Cancelada";
}

export function statusBadgeClasses(status: TaskStatus): string {
  if (status === "PLANNED") return "bg-amber-100 text-amber-800";
  if (status === "done") return "bg-green-100 text-green-800";
  return "bg-zinc-100 text-zinc-700";
}

export function isDueSoon(task: Task, hours = 12): boolean {
  if (task.status !== "PLANNED") return false;

  const now = Date.now();
  const due = new Date(task.scheduled_at).getTime();
  const diffMs = due - now;

  if (diffMs < 0) return false;
  return diffMs <= hours * 60 * 60 * 1000;
}

import type { Task } from "@/mock/tasks";

function startOfDay(d: Date) {
  return new Date(d.getFullYear(), d.getMonth(), d.getDate());
}

function endOfDay(d: Date) {
  return new Date(d.getFullYear(), d.getMonth(), d.getDate(), 23, 59, 59);
}

export function filterTasksByView(
  tasks: Task[],
  view: "day" | "week" | "month",
  reference: Date
) {
  const ref = new Date(reference);

  let from: Date;
  let to: Date;

  if (view === "day") {
    from = startOfDay(ref);
    to = endOfDay(ref);
  } else if (view === "week") {
    const day = ref.getDay(); // 0..6 (domingo..sábado)
    const start = new Date(ref);
    start.setDate(ref.getDate() - day);

    const end = new Date(start);
    end.setDate(start.getDate() + 6);

    from = startOfDay(start);
    to = endOfDay(end);
  } else {
    from = new Date(ref.getFullYear(), ref.getMonth(), 1);
    to = endOfDay(new Date(ref.getFullYear(), ref.getMonth() + 1, 0));
  }

  return tasks.filter((t) => {
    const d = new Date(t.scheduled_at);
    return d >= from && d <= to;
  });
}

export function formatPeriodLabel(
  view: "day" | "week" | "month",
  reference: Date
): string {
  const ref = new Date(reference);

  if (view === "day") {
    return ref.toLocaleDateString();
  }

  if (view === "week") {
    const day = ref.getDay();
    const start = new Date(ref);
    start.setDate(ref.getDate() - day);

    const end = new Date(start);
    end.setDate(start.getDate() + 6);

    return `${start.toLocaleDateString()} — ${end.toLocaleDateString()}`;
  }

  return ref.toLocaleDateString(undefined, { month: "long", year: "numeric" });
}

function startOfDay(d: Date) {
  return new Date(d.getFullYear(), d.getMonth(), d.getDate());
}

const today = startOfDay(new Date());

function diffInDays(a: Date, b: Date) {
  const ms =
    startOfDay(a).getTime() - startOfDay(b).getTime();
  return Math.floor(ms / (1000 * 60 * 60 * 24));
}

export function getAlertLevel(task: {
  scheduled_at: string;
  status: string;
}) {
  if (task.status !== "PLANNED") return null;

  const taskDate = new Date(task.scheduled_at);
  const daysDiff = diffInDays(taskDate, today);

  if (daysDiff < 0) return "overdue";
  if (daysDiff === 0) return "today";
  if (daysDiff === 1) return "tomorrow";
  return null;
}

export function summarizeAlerts(tasks: any[]) {
  return tasks.reduce(
    (acc, task) => {
      const level = getAlertLevel(task);
      if (!level) return acc;
      acc[level]++;
      return acc;
    },
    { overdue: 0, today: 0, tomorrow: 0 }
  );
}

  tasks: Task[];

export function TasksAlertSummary({ tasks, onClick }: Props) {
  const alerts = summarizeAlerts(tasks);

  if (
    alerts.overdue === 0 &&
    alerts.today === 0 &&
    alerts.tomorrow === 0
  ) {
    return null;
  }

  return (
    <div
      onClick={onClick}
      className="cursor-pointer rounded-lg border bg-white px-4 py-3 space-y-1 hover:bg-zinc-50"

      <h3 className="text-sm font-semibold text-zinc-900">
        Tarefas pendentes

      {alerts.overdue > 0 && (
        <div className="text-sm text-red-700">
          ⚠️ {alerts.overdue} atrasada(s)
        </div>
      )}

      {alerts.today > 0 && (
        <div className="text-sm text-yellow-700">
          🕒 {alerts.today} para hoje
        </div>
      )}

      {alerts.tomorrow > 0 && (
        <div className="text-sm text-blue-700">

// src/services/api.ts

import { ENV } from "@/lib/env";
import { getAccessToken } from "@/stores/auth";

type ApiOptions = RequestInit & {
  auth?: boolean;
};

export async function apiFetch(
  path: string,
  options: ApiOptions = {}
) {
  const token = getAccessToken();

  const headers = new Headers(options.headers || {});
  headers.set("Accept", "application/json");

  if (options.auth !== false && token) {
    headers.set("Authorization", `Bearer ${token}`);
  }

  const res = await fetch(`${ENV.API_BASE_URL}${path}`, {
    ...options,
    headers,
    credentials: "include",
  });

  if (!res.ok) {
    const text = await res.text();
    throw new Error(text || `HTTP ${res.status}`);
  }

  return res.json();
}
