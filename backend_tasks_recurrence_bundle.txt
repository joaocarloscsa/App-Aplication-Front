<?php
declare(strict_types=1);

// path: backend/src/Application/Agenda/AgendaOccurrenceGenerator.php

namespace App\Application\Agenda;

use App\Entity\AnimalAgendaItem;
use App\Entity\AgendaRecurrenceRule;

/**
 * Gera ocorrências temporais de itens de agenda (tarefas, eventos, agendamentos)
 * dentro de uma janela de tempo finita.
 *
 * ⚠️ Regra crítica:
 * - Nada é materializado no banco.
 * - Tudo é calculado sob demanda.
 * - Recorrência só existe enquanto regra + item existirem.
 */
final class AgendaOccurrenceGenerator
{
    /**
     * @return array<int, array{
     *   item: AnimalAgendaItem,
     *   startsAt: \DateTimeImmutable,
     *   endsAt: \DateTimeImmutable|null
     * }>
     */
    public function generate(
        AnimalAgendaItem $item,
        ?AgendaRecurrenceRule $rule,
        \DateTimeImmutable $from,
        \DateTimeImmutable $to
    ): array {
        // Item sem data não participa da agenda temporal
        if (!$item->getStartsAt()) {
            return [];
        }

        // Sem recorrência → ocorrência única
        if ($rule === null) {
            if ($item->getStartsAt() < $from || $item->getStartsAt() > $to) {
                return [];
            }

            return [[
                'item' => $item,
                'startsAt' => $item->getStartsAt(),
                'endsAt' => $item->getEndsAt(),
            ]];
        }

        return $this->generateRecurring(
            item: $item,
            rule: $rule,
            from: $from,
            to: $to
        );
    }

    /**
     * @return array<int, array{
     *   item: AnimalAgendaItem,
     *   startsAt: \DateTimeImmutable,
     *   endsAt: \DateTimeImmutable|null
     * }>
     */
    private function generateRecurring(
        AnimalAgendaItem $item,
        AgendaRecurrenceRule $rule,
        \DateTimeImmutable $from,
        \DateTimeImmutable $to
    ): array {
        $occurrences = [];

        $cursor = $item->getStartsAt();
        $endsAt = $item->getEndsAt();
        $until = $rule->getUntil();

        // Avança cursor até entrar na janela
        while ($cursor < $from) {
            $cursor = $this->advance($cursor, $rule);
        }

        while ($cursor <= $to) {
            if ($until && $cursor > $until) {
                break;
            }

            $occurrences[] = [
                'item' => $item,
                'startsAt' => $cursor,
                'endsAt' => $endsAt
                    ? $this->shiftEnd($item->getStartsAt(), $endsAt, $cursor)
                    : null,
            ];

            $cursor = $this->advance($cursor, $rule);
        }

        return $occurrences;
    }

    private function advance(
        \DateTimeImmutable $date,
        AgendaRecurrenceRule $rule
    ): \DateTimeImmutable {
        $value = $rule->getIntervalValue();
        $unit = $rule->getIntervalUnit();

        return match ($unit) {
            'day' => $date->modify("+{$value} days"),
            'week' => $date->modify("+{$value} weeks"),
            'month' => $date->modify("+{$value} months"),
            'year' => $date->modify("+{$value} years"),
            default => throw new \DomainException('invalid_recurrence_unit'),
        };
    }

    private function shiftEnd(
        \DateTimeImmutable $originalStart,
        \DateTimeImmutable $originalEnd,
        \DateTimeImmutable $newStart
    ): \DateTimeImmutable {
        $duration = $originalEnd->getTimestamp() - $originalStart->getTimestamp();
        return $newStart->modify("+{$duration} seconds");
    }
}

namespace App\Application\Agenda;

final class AgendaTimeWindow
{
    public static function resolve(
        string $view,
        \DateTimeImmutable $ref
    ): array {
        return match ($view) {
            'day' => [
                self::startOfDay($ref),
                self::endOfDay($ref),
            ],
            'week' => self::week($ref),
            'month' => self::month($ref),
            'year' => self::year($ref),
            default => throw new \InvalidArgumentException('invalid_view'),
        };
    }

    private static function startOfDay(\DateTimeImmutable $d): \DateTimeImmutable
    {
        return $d->setTime(0, 0);
    }

    private static function endOfDay(\DateTimeImmutable $d): \DateTimeImmutable
    {
        return $d->setTime(23, 59, 59);
    }

    private static function week(\DateTimeImmutable $d): array
    {
        $start = $d->modify('monday this week')->setTime(0, 0);
        $end = $start->modify('+6 days')->setTime(23, 59, 59);
        return [$start, $end];
    }

    private static function month(\DateTimeImmutable $d): array
    {
        $start = $d->modify('first day of this month')->setTime(0, 0);
        $end = $d->modify('last day of this month')->setTime(23, 59, 59);
        return [$start, $end];
    }

    private static function year(\DateTimeImmutable $d): array
    {
        $start = $d->setDate((int)$d->format('Y'), 1, 1)->setTime(0, 0);
        $end = $d->setDate((int)$d->format('Y'), 12, 31)->setTime(23, 59, 59);
        return [$start, $end];
    }
}

namespace App\Application\Animal\Task;

use App\Entity\AnimalAgenda;

use App\Entity\AnimalAgendaItemActionLog;
use App\Entity\AnimalAgendaItemRecurrenceRule;
use App\Entity\Person;
use Doctrine\ORM\EntityManagerInterface;

final class CreateAnimalTaskService
{
    public function __construct(
        private readonly EntityManagerInterface $em,
        private readonly GenerateRecurringAnimalTasksService $recurringGenerator
    ) {}

    public function create(
        AnimalAgenda $agenda,
        Person $actor,
        string $title,
        \DateTimeImmutable $scheduledAt,
        ?int $organizationId,
        int $priority,
        ?array $recurrence = null
    ): AnimalAgendaItem {
        $task = AnimalAgendaItem::createTask(
            agenda: $agenda,
            createdBy: $actor,
            title: $title,
            scheduledAt: $scheduledAt,
            organizationId: $organizationId,
            priority: $priority
        );

        $this->em->persist($task);

        if ($recurrence) {
            $rule = new AnimalAgendaItemRecurrenceRule(
                agendaItem: $task,
                frequency: strtoupper($recurrence['frequency']),
                intervalValue: (int) $recurrence['interval'],
                until: isset($recurrence['until'])
                    ? new \DateTimeImmutable($recurrence['until'])
                    : null
            );

            $task->setRecurrenceRule($rule);
            $this->em->persist($rule);

            $this->recurringGenerator->generate(
                agenda: $agenda,
                baseTask: $task,
                rule: $rule,
                actor: $actor,
                from: $scheduledAt,
                to: $scheduledAt->modify('+1 year')
            );
        }

        $this->em->flush();

        return $task;
    }
}

namespace App\Application\Animal\Task;

use App\Entity\AnimalAgendaItem;
use App\Entity\AnimalAgendaItemActionLog;
use App\Entity\Person;
use Doctrine\ORM\EntityManagerInterface;
use DomainException;

final class DeleteAnimalTaskService
{
    public function __construct(
        private readonly EntityManagerInterface $em
    ) {}

    public function delete(
        AnimalAgendaItem $task,
        Person $actor
    ): void {
        if ($task->getType() !== 'task') {
            throw new DomainException('only_tasks_can_be_deleted');
        }

        // Somente o criador pode deletar
        if ($task->getCreatedBy()->getId() !== $actor->getId()) {
            throw new DomainException('only_creator_can_delete_task');
        }

        // Não pode haver interações além da criação
        $logCount = $this->em->createQueryBuilder()
            ->select('COUNT(l.id)')
            ->from(AnimalAgendaItemActionLog::class, 'l')
            ->where('l.agendaItem = :item')
            ->andWhere('l.action != :created')
            ->setParameter('item', $task)
            ->setParameter('created', 'created')
            ->getQuery()
            ->getSingleScalarResult();

        if ((int) $logCount > 0) {
            throw new DomainException('task_has_interactions_and_cannot_be_deleted');
        }

        $this->em->remove($task);
        $this->em->flush();
    }
}

<?php
declare(strict_types=1);

// path: backend/src/Application/Animal/Task/GenerateRecurringAnimalTasksService.php

namespace App\Application\Animal\Task;

use App\Entity\AnimalAgenda;
use App\Entity\AnimalAgendaItem;
use App\Entity\AnimalAgendaItemRecurrenceRule;
use App\Entity\Person;
use Doctrine\ORM\EntityManagerInterface;
use DomainException;

final class GenerateRecurringAnimalTasksService
{
    public function __construct(
        private readonly EntityManagerInterface $em
    ) {}

    public function generate(
        AnimalAgenda $agenda,
        AnimalAgendaItem $baseTask,
        AnimalAgendaItemRecurrenceRule $rule,
        Person $actor,
        \DateTimeImmutable $from,
        \DateTimeImmutable $to
    ): void {
        if ($baseTask->getType() !== 'task') {
            throw new DomainException('only_tasks_can_be_recurrent');
        }

        $current = $baseTask->getStartsAt();
        if (!$current) {
            return;
        }

        // Se occurrences existir, a tarefa base já conta como 1 ocorrência.
        $remaining = $rule->getOccurrences();

        while (true) {
            // próxima data
            $current = $this->nextDate($current, $rule);

            if ($current > $to) {
                break;
            }

            if ($rule->getUntil() && $current > $rule->getUntil()) {
                break;
            }

            // cria ocorrência
            $task = AnimalAgendaItem::createTask(
                agenda: $agenda,
                createdBy: $actor,
                title: $baseTask->getTitle(),
                scheduledAt: $current,
                organizationId: null, // baseTask não expõe getter hoje
                priority: $baseTask->getPriority()
            );

            $task->updateDescription($baseTask->getDescription());

            $this->em->persist($task);

            // controla occurrences
            if ($remaining !== null) {
                $remaining--;
                if ($remaining <= 1) { // <=1 porque base já é a primeira
                    break;
                }
            }
        }

        $this->em->flush();
    }

    private function nextDate(
        \DateTimeImmutable $date,
        AnimalAgendaItemRecurrenceRule $rule
    ): \DateTimeImmutable {
        return match ($rule->getFrequency()) {
            AnimalAgendaItemRecurrenceRule::FREQ_DAILY =>
                $date->modify('+' . $rule->getIntervalValue() . ' days'),

            AnimalAgendaItemRecurrenceRule::FREQ_WEEKLY =>
                $date->modify('+' . $rule->getIntervalValue() . ' weeks'),

            AnimalAgendaItemRecurrenceRule::FREQ_MONTHLY =>
                $this->nextMonthlyDate($date, $rule),

            default => throw new DomainException('invalid_recurrence_frequency'),
        };
    }

    private function nextMonthlyDate(
        \DateTimeImmutable $date,
        AnimalAgendaItemRecurrenceRule $rule
    ): \DateTimeImmutable {
        // Se tiver dayOfMonth, fixa no mesmo dia no próximo mês (ou último dia do mês)
        if ($rule->getDayOfMonth()) {
            $next = $date->modify('first day of +' . $rule->getIntervalValue() . ' months');

            $y = (int) $next->format('Y');
            $m = (int) $next->format('m');
            $lastDay = (int) $next->format('t');
            $d = min($rule->getDayOfMonth(), $lastDay);

            return $next->setDate($y, $m, $d);
        }

        // Senão, soma meses mantendo o dia “natural”
        return $date->modify('+' . $rule->getIntervalValue() . ' months');
    }
}

namespace App\Controller\Api\V1;

use App\Application\Animal\Task\CancelAnimalTaskService;
use App\Application\Animal\Task\CreateAnimalTaskService;

use App\Application\Animal\Task\ReopenAnimalTaskService;
use App\Entity\AnimalAgendaItem;
use App\Entity\User;
use App\Repository\AnimalAgendaItemRepository;
use App\Repository\AnimalRepository;
use App\Security\Permission\PermissionCatalog;
use DomainException;
use Symfony\Bundle\FrameworkBundle\Controller\AbstractController;
use Symfony\Component\HttpFoundation\JsonResponse;
use Symfony\Component\HttpFoundation\Request;
use Symfony\Component\Routing\Attribute\Route;

#[Route('/api/v1/animals/{animalPublicId}/tasks')]
final class AnimalTaskApiController extends AbstractController
{
    public function __construct(
        private readonly AnimalRepository $animalRepository,
        private readonly AnimalAgendaItemRepository $taskRepository,
        private readonly CreateAnimalTaskService $createService,
        private readonly MarkAnimalTaskDoneService $markDoneService,
        private readonly ReopenAnimalTaskService $reopenService,
        private readonly CancelAnimalTaskService $cancelService
    ) {}

    #[Route('', methods: ['POST'])]
    public function create(string $animalPublicId, Request $request): JsonResponse
    {
        $user = $this->getUser();
        if (!$user instanceof User || !$user->getPerson()) {
            return new JsonResponse(['error' => 'unauthenticated'], 401);
        }

        $animal = $this->animalRepository->findOneBy([
            'publicId' => $animalPublicId,
        ]);

        if (!$animal) {
            return new JsonResponse(['error' => 'animal_not_found'], 404);
        }

        $this->denyAccessUnlessGranted(
            PermissionCatalog::ANIMAL_VIEW,
            $animal
        );

        $payload = json_decode($request->getContent(), true);

        if (
            !is_array($payload)
            || empty($payload['title'])
            || empty($payload['scheduled_at'])
        ) {
            return new JsonResponse(['error' => 'invalid_payload'], 400);
        }

        try {
            $scheduledAt = new \DateTimeImmutable($payload['scheduled_at']);
        } catch (\Throwable) {
            return new JsonResponse(['error' => 'invalid_date'], 400);
        }

        // ✅ PRIORIDADE COMO INT (DOMÍNIO)
        $priority = AnimalAgendaItem::PRIORITY_NORMAL;

        if (array_key_exists('priority', $payload)) {
            $priority = (int) $payload['priority'];
        }

        try {
            $task = $this->createService->create(
                $animal->getAgenda(),
                $user->getPerson(),
                (string) $payload['title'],
                $scheduledAt,
                $animal->getOrganization()?->getId(),
                $priority
            );
        } catch (\InvalidArgumentException $e) {
            return new JsonResponse(
                ['error' => $e->getMessage()],
                400
            );
        }

        return new JsonResponse(
            ['id' => $task->getId()],
            201
        );
    }

    #[Route('/{id}/done', methods: ['PATCH'])]
    public function done(string $animalPublicId, int $id, Request $request): JsonResponse
    {
        return $this->handleAction($animalPublicId, $id, $request, 'done');
    }

    #[Route('/{id}/reopen', methods: ['PATCH'])]
    public function reopen(string $animalPublicId, int $id, Request $request): JsonResponse
    {
        return $this->handleAction($animalPublicId, $id, $request, 'reopen');
    }

    #[Route('/{id}/cancel', methods: ['PATCH'])]
    public function cancel(string $animalPublicId, int $id, Request $request): JsonResponse
    {
        return $this->handleAction($animalPublicId, $id, $request, 'cancel');
    }

    private function handleAction(
        string $animalPublicId,
        int $id,
        Request $request,
        string $action
    ): JsonResponse {
        $user = $this->getUser();
        if (!$user instanceof User || !$user->getPerson()) {
            return new JsonResponse(['error' => 'unauthenticated'], 401);
        }

        $animal = $this->animalRepository->findOneBy([
            'publicId' => $animalPublicId,
        ]);

        if (!$animal) {
            return new JsonResponse(['error' => 'animal_not_found'], 404);
        }

        $this->denyAccessUnlessGranted(
            PermissionCatalog::ANIMAL_VIEW,
            $animal
        );

        $task = $this->taskRepository->find($id);

        if (
            !$task instanceof AnimalAgendaItem
            || $task->getAgenda()->animal()->getId() !== $animal->getId()
        ) {
            return new JsonResponse(['error' => 'task_not_found'], 404);
        }

        $payload = json_decode($request->getContent(), true) ?? [];
        $comment = $payload['comment'] ?? null;

        try {
            match ($action) {
                'done' => $this->markDoneService
                    ->markDone($task, $user->getPerson(), $comment),

                'reopen' => $this->reopenService
                    ->reopen($task, $user->getPerson(), $comment),

                'cancel' => $this->cancelService
                    ->cancel($task, $user->getPerson(), $comment),
            };
        } catch (DomainException $e) {
            return new JsonResponse(
                ['error' => $e->getMessage()],
                400
            );
        }

        return new JsonResponse(null, 204);
    }
}

#[Route('/api/v1/animals/{animalPublicId}/tasks/{taskId}/recurrence')]
final class AnimalTaskRecurrenceApiController extends AbstractController
{
    public function __construct(
        private readonly AnimalRepository $animalRepository,
        private readonly AnimalAgendaItemRepository $taskRepository,
        private readonly GenerateRecurringAnimalTasksService $generator,
        private readonly EntityManagerInterface $em
    ) {}

    #[Route('', methods: ['POST'])]
    public function configure(
        string $animalPublicId,
        int $taskId,
        Request $request
    ): JsonResponse {
        $user = $this->getUser();
        if (!$user instanceof User || !$user->getPerson()) {
            return new JsonResponse(['error' => 'unauthenticated'], 401);
        }

        $animal = $this->animalRepository->findOneBy(['publicId' => $animalPublicId]);
        if (!$animal) {
            return new JsonResponse(['error' => 'animal_not_found'], 404);
        }

        $this->denyAccessUnlessGranted(PermissionCatalog::ANIMAL_VIEW, $animal);

        $task = $this->taskRepository->find($taskId);
        if (!$task instanceof AnimalAgendaItem) {
            return new JsonResponse(['error' => 'task_not_found'], 404);
        }

        $payload = json_decode($request->getContent(), true);
        if (!is_array($payload) || empty($payload['frequency'])) {
            return new JsonResponse(['error' => 'invalid_payload'], 400);
        }

        try {
            $rule = new AnimalAgendaItemRecurrenceRule(
                agendaItem: $task,
                frequency: $payload['frequency'],
                intervalValue: (int)($payload['interval'] ?? 1),
                daysOfWeek: $payload['days_of_week'] ?? null,
                dayOfMonth: $payload['day_of_month'] ?? null,
                until: isset($payload['until']) ? new \DateTimeImmutable($payload['until']) : null,
                occurrences: $payload['occurrences'] ?? null
            );
        } catch (\Throwable) {
            return new JsonResponse(['error' => 'invalid_recurrence_rule'], 400);
        }

        $this->em->persist($rule);
        $this->em->flush();

        $this->generator->generate(
            agenda: $animal->getAgenda(),
            baseTask: $task,
            rule: $rule,
            actor: $user->getPerson(),
            from: new \DateTimeImmutable(),
            to: (new \DateTimeImmutable())->modify('+12 months')
        );

        return new JsonResponse(null, 201);
    }
}

namespace App\Controller\Api\V1;

#[Route('/api/v1/agenda')]
final class AgendaReadApiController extends AbstractController
{
    public function __construct(
        private readonly AnimalRepository $animalRepository,
        private readonly AnimalAgendaItemRepository $agendaItemRepository,
        private readonly AgendaRecurrenceRuleRepository $recurrenceRuleRepository,
        private readonly AgendaOccurrenceGenerator $occurrenceGenerator
    ) {}

    /**
     * Endpoint unificado de leitura da agenda.
     *
     * Retorna tarefas, eventos e agendamentos em uma linha do tempo única,
     * filtrável por janela temporal (dia, semana, mês, ano).
     */
    #[Route('/animals/{animalPublicId}', methods: ['GET'])]
    public function readForAnimal(
        string $animalPublicId,
        Request $request
    ): JsonResponse {
        $user = $this->getUser();
        if (!$user instanceof User || !$user->getPerson()) {
            return new JsonResponse(['error' => 'unauthenticated'], 401);
        }

        $animal = $this->animalRepository->findOneBy([
            'publicId' => $animalPublicId,
        ]);

        if (!$animal) {
            return new JsonResponse(['error' => 'animal_not_found'], 404);
        }

        $this->denyAccessUnlessGranted(
            PermissionCatalog::ANIMAL_VIEW,
            $animal
        );

        $fromRaw = $request->query->get('from');
        $toRaw   = $request->query->get('to');

        if (!$fromRaw || !$toRaw) {
            return new JsonResponse([
                'error' => 'from_and_to_are_required'
            ], 400);
        }

        try {
            $from = new \DateTimeImmutable($fromRaw);
            $to   = new \DateTimeImmutable($toRaw);
        } catch (\Throwable) {
            return new JsonResponse(['error' => 'invalid_date'], 400);
        }

        $items = $this->agendaItemRepository
            ->findAllForAnimalAgenda($animal->getAgenda());

        $timeline = [];

        foreach ($items as $item) {
            $rule = $this->recurrenceRuleRepository
                ->findOneBy(['agendaItem' => $item]);

            $occurrences = $this->occurrenceGenerator->generate(
                item: $item,
                rule: $rule,
                from: $from,
                to: $to
            );

            foreach ($occurrences as $occurrence) {
                $timeline[] = [
                    'type' => $item->getType(),
                    'item_id' => $item->getId(),
                    'title' => $item->getTitle(),
                    'description' => $item->getDescription(),
                    'status' => $item->getStatus(),
                    'starts_at' => $occurrence['startsAt']->format(DATE_ATOM),
                    'ends_at' => $occurrence['endsAt']
                        ? $occurrence['endsAt']->format(DATE_ATOM)
                        : null,
                ];
            }
        }

        usort(
            $timeline,
            fn ($a, $b) => strcmp($a['starts_at'], $b['starts_at'])
        );

        return new JsonResponse([
            'from' => $from->format(DATE_ATOM),
            'to'   => $to->format(DATE_ATOM),
            'items' => $timeline,
        ]);
    }
}

    }

    public function findByAnimalAndGrantedTo(Animal $animal, Person $person): array
    {
        return array_filter(
            $this->grants,
            fn (AccessGrant $grant) =>
                $this->getPrivate($grant, 'animal') === $animal
                && $this->getPrivate($grant, 'grantedTo') === $person
        );
    }

    public function save(object $grant): void
    {
        // no-op (em memória)
    }

    /* =====================================================
       Métodos auxiliares DE TESTE (não fazem parte da interface)
       ===================================================== */

    public function createActiveGrant(Animal $animal, Person $person): AccessGrant
    {
        return $this->createGrant($animal, $person, 'active');
    }

    public function createRevokedGrant(Animal $animal, Person $person): AccessGrant
    {
        return $this->createGrant($animal, $person, 'revoked');
    }

    private function createGrant(Animal $animal, Person $person, string $status): AccessGrant
    {
        $ref = new ReflectionClass(AccessGrant::class);
        $grant = $ref->newInstanceWithoutConstructor();

        $this->setPrivate($grant, 'animal', $animal);
        $this->setPrivate($grant, 'grantedTo', $person);
        $this->setPrivate($grant, 'status', $status);

        return $grant;
    }

    private function setPrivate(object $object, string $property, mixed $value): void
    {
        $ref = new ReflectionClass($object);

        while (!$ref->hasProperty($property)) {
            $ref = $ref->getParentClass();
            if ($ref === false) {
                throw new \LogicException("Property {$property} not found");
            }
        }

        $prop = $ref->getProperty($property);
        $prop->setAccessible(true);
        $prop->setValue($object, $value);
    }

    private function getPrivate(object $object, string $property): mixed
    {
        $ref = new ReflectionClass($object);

        while (!$ref->hasProperty($property)) {
            $ref = $ref->getParentClass();
            if ($ref === false) {
                throw new \LogicException("Property {$property} not found");
            }
        }

        $prop = $ref->getProperty($property);
        $prop->setAccessible(true);

        return $prop->getValue($object);
    }
}

/**
 * Garante a REGRA FUNDAMENTAL:
 *
 * - Agenda NÃO acompanha o cão
 * - Ao aceitar transferência:
 *   - agenda antiga fica congelada (snapshot)
 *   - nova agenda começa vazia para o novo tutor
 */
final class AcceptAnimalTransferAgendaIsolationTest extends WebTestCase
{
    private EntityManagerInterface $em;
    private AnimalTransferRepository $transferRepo;
    private EntityRepository $snapshotRepo;
    protected function setUp(): void
    {
 

        self::markTestSkipped('Agenda do animal ainda não esta implementada no dominio');
	parent::setUp();
        $this->em = self::getContainer()->get(EntityManagerInterface::class);
        $this->transferRepo = self::getContainer()->get(AnimalTransferRepository::class);
        $this->snapshotRepo = $this->em->getRepository(AnimalOwnershipSnapshot::class);

    }

    public function testAgendaIsIsolatedAfterAcceptingTransfer(): void
    {
        $ctx = $this->seedTransferWithAgendaHistory();

        $client = static::createClient();
        $client->loginUser($ctx['toUser']);

        $client->request(
            'POST',
            '/api/v1/animal-transfers/' . $ctx['transferPublicId'] . '/accept'
        );

        self::assertResponseStatusCodeSame(200);

        /** @var AnimalTransfer $transfer */
        $transfer = $this->transferRepo->findOneBy(['publicId' => $ctx['transferPublicId']]);
        self::assertSame('accepted', $transfer->getStatus());

        /** @var Animal $animal */
        $animal = $this->em->getRepository(Animal::class)->find($ctx['animalId']);
        self::assertNotNull($animal);

        // Tutor foi alterado
        self::assertSame(
            $ctx['toPerson']->getId(),
            $animal->getTutor()?->getId()
        );

        // Snapshot criado (agenda antiga congelada)
        $snapshots = $this->snapshotRepo->findBy(['animal' => $animal]);
        self::assertCount(1, $snapshots);

        $snapshot = $snapshots[0];
        $data = $snapshot->getSnapshot();

        self::assertArrayHasKey('agenda', $data);
        self::assertCount(2, $data['agenda']); // histórico antigo

        // Agenda ATUAL do animal deve estar vazia
        self::assertCount(
            0,
            $animal->getAgendaEntries(),
            'Agenda nova deve iniciar vazia após transferência'
        );
    }

    {

        $this->em->persist($fromUser);

        self::assertNotNull($animalId);

        $transfer = new AnimalTransfer(
            'TRF_ACCEPT_'.bin2hex(random_bytes(4)),
            $animal,
            $fromUser,
            $fromPerson,
            null,
            $toPerson,
            null,
            'accept-test'
        );

        $em->persist($transfer);
        $em->flush();

        $client->loginUser($toUser);

        self::assertResponseIsSuccessful();

        $reloaded = $em->getRepository(Animal::class)->find($animalId);
